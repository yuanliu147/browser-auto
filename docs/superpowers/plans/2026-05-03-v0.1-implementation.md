# browser-auto v0.1 实施计划

> **给执行 Agent:** 使用 superpowers:subagent-driven-development 逐任务实施。步骤使用 checkbox (`- [ ]`) 语法跟踪进度。

**目标:** 构建 browser-auto v0.1 骨架 —— 带分层记忆（L3/L2）的浏览器自动化工具，提供 act/extract/observe API 和 7 个扩展钩子。

**架构:** Playwright 驱动 + 4 层记忆降级链（L3 动作缓存 → L2 定位器缓存 → L1 区域占位 → L0 全量 LLM）。核心流程：用户在 EnhancedPage（Playwright Page 的 Proxy）上调用 `act()`/`extract()`/`observe()`，经过指纹计算 → 记忆查找 → 规划器（LLM）→ 执行器 → 追踪管道。所有扩展点（SnapshotStrategy、ActionHook、TraceSink、CDP attach）在 v0.1 定义为接口。

**技术栈:** TypeScript strict, Playwright, pnpm, tsup (esm+cjs+d.ts), vitest, ai-sdk + @ai-sdk/openai, zod, debug, biome

**当前状态:** Step 1（架构文档）已完成。项目根目录有裸 `package.json`（pnpm init 已完成）。尚无 src/ 目录。

```ts
import type { Page } from "playwright";
import type { InteractiveNode } from "../types";
import type { SnapshotStrategy } from "./index";
import { createHash } from "node:crypto";

const INTERACTIVE_ROLES = new Set([
  "button",
  "link",
  "textbox",
  "combobox",
  "checkbox",
  "menuitem",
  "tab",
  "menu",
  "listbox",
  "option",
  "radio",
  "switch",
  "slider",
  "spinbutton",
  "searchbox",
]);

type A11yNode = {
  role: string;
  name: string;
  children?: A11yNode[];
};

export class A11ySnapshotStrategy implements SnapshotStrategy {
  name = "a11y";

  async getInteractiveNodes(page: Page): Promise<InteractiveNode[]> {
    const snapshot = await page.accessibility.snapshot();
    if (!snapshot) return [];

    const result: InteractiveNode[] = [];
    this.collectInteractive(snapshot as A11yNode, [], result);
    return result;
  }

  private collectInteractive(
    node: A11yNode,
    parentChain: string[],
    result: InteractiveNode[]
  ): void {
    const currentChain = [...parentChain, node.role];

    if (INTERACTIVE_ROLES.has(node.role)) {
      result.push({
        role: node.role,
        name: this.normalizeName(node.name),
        parentChain: parentChain.slice(),
        attributes: {},
      });
    }

    if (node.children) {
      for (const child of node.children) {
        this.collectInteractive(child, currentChain, result);
      }
    }
  }

  /** Replace dynamic values like numbers with placeholders for stable hashing */
  private normalizeName(name: string): string {
    return name.replace(/\d+/g, "${num}");
  }

  fingerprint(nodes: InteractiveNode[]): string {
    const sorted = [...nodes].sort((a, b) => {
      const keyA = `${a.role}:${a.name}:${a.parentChain.join(">")}`;
      const keyB = `${b.role}:${b.name}:${b.parentChain.join(">")}`;
      return keyA.localeCompare(keyB);
    });

    const content = sorted
      .map((n) => `${n.role}:${n.name}:${n.parentChain.join(">")}`)
      .join("\n");

    return createHash("sha256").update(content).digest("hex");
  }
}
```

```ts
import type { Page } from "playwright";
import type { Fingerprint, InteractiveNode, SnapshotStrategy } from "../types";
import { A11ySnapshotStrategy } from "../snapshot/a11y";

export async function computeFingerprint(
  page: Page,
  strategy?: SnapshotStrategy
): Promise<Fingerprint> {
  const s = strategy ?? new A11ySnapshotStrategy();
  const nodes = await s.getInteractiveNodes(page);
  const domHash = s.fingerprint(nodes);

  // Build a compact summary string
  const domSummary = buildSummary(nodes);

  return {
    url: page.url(),
    domHash,
    domSummary,
  };
}

function buildSummary(nodes: InteractiveNode[]): string {
  const parts: string[] = [];
  for (const node of nodes) {
    const parentStr =
      node.parentChain.length > 0 ? node.parentChain.join(">") + ">" : "";
    parts.push(`${parentStr}${node.role}.${node.name || "?"}`);
  }
  return parts.join(",");
}

export function hashNodes(nodes: InteractiveNode[]): string {
  return new A11ySnapshotStrategy().fingerprint(nodes);
}
```

```ts
export type ActionRecord = {
  id: string;
  instruction: string;

export function matchAction(
  instruction: string,
  fp: string,
  records: ActionRecord[]
): ActionRecord | null {
  for (const record of records) {
    if (
      record.instruction === instruction &&
      record.fingerprint.domHash === fp
    ) {
      return record;
    }
  }
  return null;
}

export function createActionRecord(
  instruction: string,
  domHash: string,
  domSummary: string,
  url: string,
  plan: ActionStep[]
): ActionRecord {
  const now = new Date().toISOString();
  return {
    id: makeId("act"),
    instruction,
    fingerprint: { url, domHash, domSummary },
    plan,
    stats: {
      createdAt: now,
      lastUsedAt: now,
      hitCount: 0,
      fallbackCount: 0,
    },
  };
}

export function touchActionRecord(record: ActionRecord): void {
  record.stats.lastUsedAt = new Date().toISOString();
  record.stats.hitCount++;
```

```ts
import { readFileSync, writeFileSync, existsSync, mkdirSync } from "node:fs";
import { dirname } from "node:path";
import type { ActionStep, LocatorEntry, Memory, RegionEntry } from "../types";
import { MemoryFileSchema, type MemoryFile } from "./schema";
import {
  createActionRecord,
  matchAction,
  touchActionRecord,
  type ActionRecord,
} from "./action-memory";
import {
  createLocatorRecord,
  matchLocator,
  type LocatorRecord,
} from "./locator-memory";

function emptyMemoryFile(): MemoryFile {
  return { version: 1, actions: [], locators: [], regions: [] };
}

export class FileMemory implements Memory {
  constructor(private filePath: string) {}

  // L3: Action Plan
  recallPlan(instruction: string, fp: string): ActionStep[] | null {
    const data = this.readMemoryFile();
    const hit = matchAction(instruction, fp, data.actions);
    if (hit) {
      touchActionRecord(hit);
      this.writeMemoryFile(data);
      return hit.plan;
    }
    return null;
  }

  rememberPlan(
    instruction: string,
    fp: string,
    plan: ActionStep[],
    url?: string
  ): void {
    const data = this.readMemoryFile();
    data.actions.push(createActionRecord(instruction, fp, "", url ?? "", plan));
    this.writeMemoryFile(data);
  }

  invalidatePlan(instruction: string, fp: string): void {
    const data = this.readMemoryFile();
    data.actions = data.actions.filter(
      (a) => !(a.instruction === instruction && a.fingerprint.domHash === fp)
    );
    this.writeMemoryFile(data);
  }

  // L2: Locator
  recallLocator(semanticKey: string, fp: string): LocatorEntry | null {
    const data = this.readMemoryFile();
    const hit = matchLocator(semanticKey, fp, data.locators);
    if (hit) {
      hit.entry.successCount++;
      this.writeMemoryFile(data);
      return hit.entry;
    }
    return null;
  }

  rememberLocator(
    semanticKey: string,
    fp: string,
    entry: LocatorEntry,
    url?: string
  ): void {
    const data = this.readMemoryFile();
    data.locators.push(createLocatorRecord(semanticKey, fp, url ?? "", entry));
    this.writeMemoryFile(data);
  }

  invalidateLocator(semanticKey: string, fp: string): void {
    const data = this.readMemoryFile();
    data.locators = data.locators.filter(
      (l) => !(l.semanticKey === semanticKey && l.fingerprint.domHash === fp)
    );
    this.writeMemoryFile(data);
  }

  // L1: Region — not implemented in v0.1
  recallRegion(_semanticKey: string, _fp: string): RegionEntry | null {
    throw new Error("L1 RegionMemory is not implemented in v0.1");
  }

  rememberRegion(_semanticKey: string, _fp: string, _entry: RegionEntry): void {
    throw new Error("L1 RegionMemory is not implemented in v0.1");
  }

  // Internal
  private readMemoryFile(): MemoryFile {
    try {
      if (existsSync(this.filePath)) {
        const raw = readFileSync(this.filePath, "utf-8");
        const parsed = JSON.parse(raw);
        return MemoryFileSchema.parse(parsed);
      }
    } catch {
      // Corrupt or missing — start fresh
    }
    return emptyMemoryFile();
  }

  private writeMemoryFile(data: MemoryFile): void {
    const dir = dirname(this.filePath);
    if (!existsSync(dir)) {
      mkdirSync(dir, { recursive: true });
    }
    writeFileSync(this.filePath, JSON.stringify(data, null, 2));
  }
}
```

```ts
/**
 * Mock LLM for unit tests — maps specific inputs to predefined outputs.
 * Never makes real API calls.
 */
export class MockLLM implements LLMClient {
  private planResponses = new Map<string, ActionStep[]>();
  private locateResponses = new Map<string, Record<string, LocatorEntry>>();
  private observeResponses = new Map<string, NodeRef[]>();

  /** Register a mock plan response keyed by snapshot+instruction combined */
  registerPlan(
    snapshotContains: string,
    instruction: string,
    plan: ActionStep[]
  ): void {
    this.planResponses.set(this.key(snapshotContains, instruction), plan);
  }

  async plan(snapshot: string, instruction: string): Promise<PlanResult> {
    const key = this.key(snapshot, instruction);
    const plan = this.planResponses.get(key);
    if (!plan) {
      // Try partial match by checking if snapshot contains registered substring
      for (const [k, v] of this.planResponses) {
        if (snapshot.includes(k) && instruction === k.split("||")[1]) {
          return { plan: v, usage: { promptTokens: 0, completionTokens: 0 } };
        }
      }
      throw new Error(
        `MockLLM: no plan registered for instruction "${instruction}"`
      );
    }
    return { plan, usage: { promptTokens: 100, completionTokens: 50 } };
  }

  registerLocate(field: string, entry: LocatorEntry): void {
    this.locateResponses.set(field, { [field]: entry });
  }

  async locate(_snapshot: string, fields: string[]): Promise<LocatorResult> {
    const locators: Record<string, LocatorEntry> = {};
    for (const field of fields) {
      const entry = this.locateResponses.get(field);
      if (entry) {
        locators[field] = entry[field];
      } else {
        locators[field] = {
          selectors: [],
          hint: field,
          selectorHits: [],
          successCount: 0,
          failCount: 0,
        };
      }
    }
    return { locators, usage: { promptTokens: 50, completionTokens: 20 } };
  }

  registerObserve(instruction: string, nodes: NodeRef[]): void {
    this.observeResponses.set(instruction, nodes);
  }

  async observe(
    _snapshot: string,
    instruction: string
  ): Promise<ObserveResult> {
    const nodes = this.observeResponses.get(instruction) ?? [];
    return { nodes, usage: { promptTokens: 30, completionTokens: 10 } };
  }

  private key(snapshot: string, instruction: string): string {
    return `${snapshot.slice(0, 50)}||${instruction}`;
  }
}
```

```ts
import type { Page } from "playwright";
import type { NodeRef } from "../types";

export async function resolveSelector(
  page: Page,
  nodeRef: NodeRef
): Promise<{ selector: string; index: number } | null> {
  for (let i = 0; i < nodeRef.selectors.length; i++) {
    const selector = nodeRef.selectors[i];
    try {
      const count = await page.locator(selector).count();
      if (count > 0) {
        return { selector, index: i };
      }
    } catch {
      // Invalid selector syntax — skip to next
      continue;
    }
  }
  return null;
}
```

```ts
import type { Page } from "playwright";
import type {
  ActionStep,
  ActionResult,
  BeforeActionInterceptor,
  AfterActionInterceptor,
  TraceSink,
} from "../types";
import { resolveSelector } from "./selector-resolver";
import { createInterceptorChain } from "./interceptor";
import { executorDebug } from "../debug";

export async function executeSteps(
  page: Page,
  steps: ActionStep[],
  opts: {
    sessionId: string;
    instruction: string;
    befores?: BeforeActionInterceptor[];
    afters?: AfterActionInterceptor[];
    trace?: TraceSink[];
  }
): Promise<ActionResult[]> {
  const chain = createInterceptorChain(opts.befores ?? [], opts.afters ?? []);
  const results: ActionResult[] = [];

  for (const step of steps) {
    const ctx = {
      sessionId: opts.sessionId,
      page,
      step,
      instruction: opts.instruction,
    };

    const abort = await chain.runBefore(ctx);
    if (abort?.abort) {
      results.push({
        success: false,
        selectorIndex: -1,
        durationMs: 0,
        error: new Error(abort.reason),
      });
      continue;
    }

    const start = Date.now();
    let result: ActionResult;

    try {
      const resolved = await resolveSelector(page, step.target);
      if (!resolved) {
        throw new Error(
          `No selector matched for ${step.kind} on "${step.target.name}"`
        );
      }

      await executeSingleStep(page, step, resolved.selector);

      result = {
        success: true,
        selectorIndex: resolved.index,
        durationMs: Date.now() - start,
      };
    } catch (err) {
      result = {
        success: false,
        selectorIndex: -1,
        durationMs: Date.now() - start,
        error: err as Error,
      };
    }

    results.push(result);
    await chain.runAfter(ctx, result);
  }

  return results;
}

async function executeSingleStep(
  page: Page,
  step: ActionStep,
  selector: string
): Promise<void> {
  switch (step.kind) {
    case "click":
      await page.locator(selector).first().click();
      break;
    case "fill": {
      const value = typeof step.value === "string" ? step.value : "";
      await page.locator(selector).first().fill(value);
      break;
    }
    case "press":
      await page.keyboard.press(step.key);
      break;
    case "select":
      await page.locator(selector).first().selectOption(step.value);
      break;
    case "hover":
      await page.locator(selector).first().hover();
      break;
    case "wait": {
      if (step.condition === "visible" && selector) {
        await page
          .locator(selector)
          .first()
          .waitFor({ state: "visible", timeout: step.ms ?? 5000 });
      } else if (step.condition === "timeout" && step.ms) {
        await page.waitForTimeout(step.ms);
      } else {
        await page.waitForLoadState("networkidle");
      }
      break;
    }
  }
}
```

```ts
function createActFn(
  page: Page,
  sessionId: string,
  pageId: string,
  llm: ReturnType<typeof createLLMClient>,
  snapshotter: A11ySnapshotStrategy,
  memory: Memory,
  traceBus: TraceBus,
  befores?: BeforeActionInterceptor[],
  afters?: AfterActionInterceptor[]
) {
  return async function act(
    instruction: string,
    vars?: Record<string, string>
  ): Promise<void> {
    const startTs = Date.now();
    await traceBus.emit({
      type: "act.start",
      sessionId,
      pageId,
      instruction,
      ts: startTs,
    });

    // Substitute variables in instruction
    let resolvedInstruction = instruction;
    if (vars) {
      for (const [key, value] of Object.entries(vars)) {
        resolvedInstruction = resolvedInstruction.replace(`\${${key}}`, value);
      }
    }

    // Compute fingerprint for the current page
    const fp = await computeFingerprint(page, snapshotter);

    // L3: Try recall from memory
    const cachedPlan = memory.recallPlan(instruction, fp.domHash);
    if (cachedPlan) {
      memoryDebug("L3 hit for instruction: %s", instruction);
      await traceBus.emit({
        type: "memory.hit",
        sessionId,
        layer: "L3",
        key: fp.domHash.slice(0, 16),
        ts: Date.now(),
      });

      // Self-healing: verify first step's target still exists
      const results = await executeSteps(page, cachedPlan, {
        sessionId,
        instruction: resolvedInstruction,
        befores,
        afters,
        trace: [],
      });

      const allSuccess = results.every((r) => r.success);
      if (allSuccess) {
        await traceBus.emit({
          type: "act.end",
          sessionId,
          success: true,
          durationMs: Date.now() - startTs,
          ts: Date.now(),
        });
        return;
      }

      memoryDebug("L3 stale, invalidating and falling back to L0");
      memory.invalidatePlan(instruction, fp.domHash);
      await traceBus.emit({
        type: "memory.invalidate",
        sessionId,
        layer: "L3",
        key: fp.domHash.slice(0, 16),
        reason: "self-healing check failed",
        ts: Date.now(),
      });
    } else {
      await traceBus.emit({
        type: "memory.miss",
        sessionId,
        layer: "L3",
        key: fp.domHash.slice(0, 16),
        ts: Date.now(),
      });
    }

    // L0: Full LLM planning
    const nodes = await snapshotter.getInteractiveNodes(page);
    const snapshotSize = nodes.length;
    await traceBus.emit({
      type: "llm.request",
      sessionId,
      snapshotSize,
      model: "llm",
      ts: Date.now(),
    });

    const { plan, usage } = await planActions(llm, nodes, resolvedInstruction);
    await traceBus.emit({
      type: "llm.response",
      sessionId,
      plan,
      usage,
      ts: Date.now(),
    });

    // Execute
    for (let i = 0; i < plan.length; i++) {
      const step = plan[i];
      await traceBus.emit({
        type: "step.start",
        sessionId,
        step,
        attempt: 1,
        ts: Date.now(),
      });
      const results = await executeSteps(page, [step], {
        sessionId,
        instruction: resolvedInstruction,
        befores,
        afters,
        trace: [],
      });
      const result = results[0];
      if (result.success) {
        await traceBus.emit({
          type: "step.success",
          sessionId,
          step,
          selectorIndex: result.selectorIndex,
          ts: Date.now(),
        });
      } else {
        await traceBus.emit({
          type: "step.fail",
          sessionId,
          step,
          error: result.error?.message ?? "unknown",
          ts: Date.now(),
        });
      }
    }

    // Remember for next time
    memory.rememberPlan(instruction, fp.domHash, plan, fp.url);
    await traceBus.emit({
      type: "memory.write",
      sessionId,
      layer: "L3",
      key: fp.domHash.slice(0, 16),
      ts: Date.now(),
    });
    await traceBus.emit({
      type: "act.end",
      sessionId,
      success: true,
      durationMs: Date.now() - startTs,
      ts: Date.now(),
    });
  };
}

function createExtractFn(
  page: Page,
  sessionId: string,
  pageId: string,
  llm: ReturnType<typeof createLLMClient>,
  snapshotter: A11ySnapshotStrategy,
  memory: Memory,
  traceBus: TraceBus
) {
  return async function extract<T>(opts: {
    schema: z.ZodType<T>;
    instruction?: string;
  }): Promise<T> {
    const startTs = Date.now();

    // Extract field keys from zod schema
    let fieldKeys: string[] = [];
    if (opts.schema instanceof z.ZodObject) {
      fieldKeys = Object.keys(opts.schema.shape);
    }

    await traceBus.emit({
      type: "extract.start",
      sessionId,
      pageId,
      schemaFields: fieldKeys,
      ts: startTs,
    });

    const fp = await computeFingerprint(page, snapshotter);
    const result: Record<string, unknown> = {};
    const missedFields: string[] = [];

    for (const field of fieldKeys) {
      const semanticKey = `extract:${field}`;

      // L2: Try cached locator
      const cached = memory.recallLocator(semanticKey, fp.domHash);
      if (cached && cached.selectors.length > 0) {
        memoryDebug("L2 hit for field: %s", field);
        await traceBus.emit({
          type: "memory.hit",
          sessionId,
          layer: "L2",
          key: semanticKey,
          ts: Date.now(),
        });

        try {
          const el = page.locator(cached.selectors[0]).first();
          const text = await el.textContent();
          if (text && text.trim()) {
            result[field] = text.trim();
            continue;
          }
        } catch {
          memoryDebug("L2 stale for field: %s", field);
          memory.invalidateLocator(semanticKey, fp.domHash);
          await traceBus.emit({
            type: "memory.invalidate",
            sessionId,
            layer: "L2",
            key: semanticKey,
            reason: "selector returned empty",
            ts: Date.now(),
          });
        }
      } else {
        await traceBus.emit({
          type: "memory.miss",
          sessionId,
          layer: "L2",
          key: semanticKey,
          ts: Date.now(),
        });
      }

      missedFields.push(field);
    }

    // L0: LLM for missed fields
    if (missedFields.length > 0) {
      const nodes = await snapshotter.getInteractiveNodes(page);
      const { locators } = await planLocators(llm, nodes, missedFields);

      for (const field of missedFields) {
        const entry = locators[field];
        if (entry && entry.selectors.length > 0) {
          const semanticKey = `extract:${field}`;
          memory.rememberLocator(semanticKey, fp.domHash, entry, fp.url);
          await traceBus.emit({
            type: "memory.write",
            sessionId,
            layer: "L2",
            key: semanticKey,
            ts: Date.now(),
          });

          try {
            const text = await page
              .locator(entry.selectors[0])
              .first()
              .textContent();
            result[field] = text?.trim() ?? "";
          } catch {
            result[field] = "";
          }
        }
      }
    }

    await traceBus.emit({
      type: "extract.end",
      sessionId,
      success: true,
      durationMs: Date.now() - startTs,
      ts: Date.now(),
    });

    return opts.schema.parse(result) as T;
  };
}

function createObserveFn(
  page: Page,
  llm: ReturnType<typeof createLLMClient>,
  snapshotter: A11ySnapshotStrategy
) {
  return async function observe(instruction: string): Promise<NodeRef[]> {
    const nodes = await snapshotter.getInteractiveNodes(page);
    const snapshot = nodes
      .map((n) => `[${n.role}] ${n.name} (${n.parentChain.join(">")})`)
      .join("\n");

    const result = await llm.observe(snapshot, instruction);
    return result.nodes;
  };
}
```

```ts
    browser = await chromium.connectOverCDP(opts.browser.cdpEndpoint);
  } else if (opts.browser.mode === "persistent") {
    const context = await chromium.launchPersistentContext(
      opts.browser.userDataDir,
      opts.browser.options ?? {}
    );
    // launchPersistentContext returns a BrowserContext, not Browser — wrap
    browser = context as unknown as Browser;
  } else {
    throw new Error(`Unknown browser mode: ${(opts.browser as any).mode}`);
  }

  let closed = false;
  const stats: SessionStats = {
    l3HitCount: 0,
    l2HitCount: 0,
    llmCallCount: 0,
    totalActions: 0,
  };

  return {
    async newPage(): Promise<EnhancedPage> {
      if (closed) throw new Error("Session is closed");
      const rawPage = await browser.newPage();
      return enhancePage(rawPage, {
        llm: opts.llm,
        memory,
        sessionId,
        befores: opts.policies,
        traceSinks,
      });
    },

    async close(): Promise<void> {
      closed = true;
      await browser.close();
    },

    getStats(): SessionStats {
      return stats;
    },
  };
}
```

```ts
/**
 * Example 05: Custom SnapshotStrategy
 *
 * Demonstrates injecting a custom snapshot strategy that only
 * returns elements with specific data-testid attributes.
 *
 * Usage: OPENAI_API_KEY=sk-xxx pnpm tsx examples/05-custom-snapshot.ts
 */
import { chromium } from "playwright";
import { createSession } from "../src/session";
import type {
  Page,
  SnapshotStrategy,
  InteractiveNode,
  NodeRef,
} from "../src/types";
import { createHash } from "node:crypto";

class DataTestIdOnlyStrategy implements SnapshotStrategy {
  name = "data-testid-only";

  async getInteractiveNodes(page: Page): Promise<InteractiveNode[]> {
    // Only find elements with data-testid attribute
    const elements = await page.$$eval("[data-testid]", (els) =>
      els.map((el) => ({
        role: el.getAttribute("role") || el.tagName.toLowerCase(),
        name: el.getAttribute("aria-label") || el.textContent?.trim() || "",
        parentChain: [] as string[],
        attributes: { "data-testid": el.getAttribute("data-testid") || "" },
      }))
    );
    return elements;
  }

  fingerprint(nodes: InteractiveNode[]): string {
    const content = nodes
      .map((n) => `${n.attributes["data-testid"]}`)
      .sort()
      .join("\n");
    return createHash("sha256").update(content).digest("hex");
  }
}

async function main() {
  const apiKey = process.env.OPENAI_API_KEY;
  if (!apiKey) {
    console.error("Error: OPENAI_API_KEY is required.");
    process.exit(1);
  }

  console.log("Using custom DataTestIdOnlyStrategy...");
  // The custom strategy is demonstrated via the SnapshotStrategy interface.
  // In v0.1, you can use it directly with computeFingerprint and planner functions.
  // Full injection into createSession is a v0.3 feature (L1 region support).
  console.log("Custom strategy interface verified.");
  console.log("Strategy name:", new DataTestIdOnlyStrategy().name);
}

main().catch(console.error);
```
